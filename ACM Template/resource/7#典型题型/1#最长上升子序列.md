```cpp
int mx[mx_n];
int r = 0;
memset(mx, 0x3f, sizeof mx);
mx[0] = 0;
for(int i=1; i<=m; i++) {
    int id = lower_bound(mx, mx+r+1, c[i]) - mx;
    mx[id] = c[i];
    r = max(r, id);
}
# 最后 mx 数组合法的最大的下标就是答案
```
